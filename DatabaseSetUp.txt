//PostgreSQL:
1.Set Up Database:
-- Database: PROJECT_PIPELINE

-- DROP DATABASE IF EXISTS "PROJECT_PIPELINE";

CREATE DATABASE "PROJECT_PIPELINE"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'English_United States.1252'
    LC_CTYPE = 'English_United States.1252'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;

GRANT TEMPORARY, CONNECT ON DATABASE "PROJECT_PIPELINE" TO PUBLIC;

GRANT CONNECT ON DATABASE "PROJECT_PIPELINE" TO etl;

GRANT ALL ON DATABASE "PROJECT_PIPELINE" TO postgres;

2. Create role:
CREATE USER etl WITH PASSWORD 'demopass';
GRANT CONNECT ON DATABASE "PROJECT_PIPELINE" TO etl;
GRANT ALL ON ALL TABLES IN SCHEMA public TO etl;

//MSSQL Server:
1. Create user:
USE [master]
GO
CREATE LOGIN [etl] WITH PASSWORD =N'demopass', DEFAULT_DATABASE = [PROJECT_PIPELINE], CHECK_EXPIRATION = OFF, CHECK_POLICY = OFF
GO
USE [PROJECT_PIPELINE]
GO
CREATE USER [etl] FOR LOGIN [etl]
GO
USE [PROJECT_PIPELINE]
ALTER ROLE [db_datareader] ADD MEMBER [etl]
GO
USE [master]
GO 
GRANT CONNECT SQL TO [etl]
GO
